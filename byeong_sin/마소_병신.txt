# 💀 관리자 권한 필수!
Write-Host "Microsoft 백그라운드 똥서비스 제거 시작..." -ForegroundColor Red

$servicesToDisable = @(
    "DiagTrack",                      # Connected User Experience / Telemetry
    "SysMain",                        # Superfetch, 램 과다 점유
    "WSearch",                        # Windows Search
    "XblGameSave",                    # Xbox Game Save
    "XboxNetApiSvc",                 # Xbox 네트워크
    "XblAuthManager",                # Xbox Auth
    "OneSyncSvc",                    # Outlook / 일정 동기화
    "CDPSvc",                         # Connected Devices Platform Service
    "MapsBroker",                    # 지도 관련
    "Fax",                           # 누구 fax 씀?
    "RetailDemo"                     # 데모모드 ㅋㅋ
)

foreach ($svc in $servicesToDisable) {
    Get-Service -Name $svc -ErrorAction SilentlyContinue | ForEach-Object {
        if ($_.Status -ne 'Stopped') {
            Stop-Service -Name $_.Name -Force
        }
        Set-Service -Name $_.Name -StartupType Disabled
        Write-Host "$($_.Name) 꺼지고 비활성화 완료" -ForegroundColor Green
    }
}

Write-Host "불필요한 서비스들 전부 제거 완료." -ForegroundColor Yellow
